아래 순서대로 설정하면 수강신청 · 수료 정보가 구글 스프레드시트에 자동으로 저장됩니다. 최초 1회만 설정하면 됩니다.

  1. 구글 스프레드시트 ▶ 열기
  2. 상단 메뉴 [확장 프로그램] → [Apps Script] 클릭
  3. 기존 코드를 지우고 아래 코드를 전체 붙여넣기 후 Ctrl+S 저장
  4. 상단 [배포] → [새 배포] 클릭 → 유형: 웹앱 선택
  5. 실행 계정: 나 / 액세스 권한: 모든 사용자 → [배포]
  6. 구글 계정 권한 허용 후, 생성된 웹앱 URL을 아래에 붙여넣기
const SHEET_ID = '1WZc6VnTsGranw5jBmHpTpj0jfB-D3qU3iUvi1WvtpWg';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('수강생명단');
    if (!sheet) {
      sheet = ss.insertSheet('수강생명단');
      const hdr = sheet.getRange(1,1,1,8);
      sheet.appendRow(['No','시설명','지역','성명','생년월일','직위','신청일시','수료여부']);
      sheet.getRange(1,1,1,8).setFontWeight('bold')
        .setBackground('#1a2744').setFontColor('#ffffff');
    }
    if (data.action === 'register') {
      const no = sheet.getLastRow(); // 헤더 포함 행수 = 순번
      sheet.appendRow([no, data.facility, data.region, data.name,
                       data.birth, data.position, data.registeredAt, '수강대기']);
      return out({ok:true, row: sheet.getLastRow()});
    }
    if (data.action === 'complete') {
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][3] == data.name && rows[i][4] == data.birth) {
          sheet.getRange(i+1,8).setValue('교육수료')
            .setBackground('#d1e7dd').setFontColor('#0f5132').setFontWeight('bold');
          return out({ok:true});
        }
      }
      return out({ok:false, msg:'행 없음'});
    }
    if (data.action === 'list') {
      const rows = sheet.getDataRange().getValues();
      return out({ok:true, rows: rows.slice(1)});
    }
    return out({ok:false});
  } catch(err) {
    return out({ok:false, err: err.toString()});
  }
}

function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) { return doPost(e); }
