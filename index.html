📊 Google Sheets 수료 관리 최적화 스크립트

구글 스프레드시트는 최종 수료 여부만 추적하도록 코드가 최적화되었습니다. 아래 코드로 새 배포를 진행해주시면 더욱 안정적으로 동작합니다.

  1. 구글 스프레드시트 [확장 프로그램] → [Apps Script] 클릭
  2. 기존 코드를 모두 삭제하고 아래 코드를 복사해서 붙여넣기
  3. 상단 [배포] → [새 배포] 혹은 [배포 관리]에서 신규 버전 배포 완료
const SHEET_ID = '1WZc6VnTsGranw5jBmHpTpj0jfB-D3qU3iUvi1WvtpWg';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    
    let sheet = ss.getSheetByName('수강생명단');
    if (!sheet) {
      sheet = ss.insertSheet('수강생명단');
      sheet.appendRow(['No','시설명','지역','성명','생년월일','직위','신청일시','최종상태']);
      sheet.getRange(1,1,1,8).setFontColor('#ffffff').setBackground('#1a2744');
    }
    
    if (data.action === 'register_or_login') {
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][3] == data.name && rows[i][4] == data.birth) {
          return out({
            ok: true, isNew: false, student: {
              no: rows[i][0], facility: rows[i][1], region: rows[i][2],
              name: rows[i][3], birth: rows[i][4], position: rows[i][5], registeredAt: rows[i][6],
              status: rows[i][7] || '수강대기', completed: String(rows[i][7]).includes('교육수료')
            }
          });
        }
      }
      const no = sheet.getLastRow(); 
      sheet.appendRow([no, data.facility, data.region, data.name, data.birth, data.position, data.registeredAt, '수강대기']);
      return out({ ok: true, isNew: true, student: { name: data.name, birth: data.birth, status: '수강대기', completed: false } });
    }
    
    if (data.action === 'update_status') {
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][3] == data.name && rows[i][4] == data.birth) {
          const rowNum = i + 1;
          sheet.getRange(rowNum, 8).setValue(data.status);
          if (data.status === '교육수료') {
            sheet.getRange(rowNum, 8).setBackground('#d1e7dd').setFontColor('#0f5132');
          }
          return out({ok:true});
        }
      }
      return out({ok:false, msg:'사용자 없음'});
    }
    
    if (data.action === 'list') {
      const rows = sheet.getDataRange().getValues();
      return out({ok:true, rows: rows.slice(1)});
    }
    return out({ok:false});
  } catch(err) {
    return out({ok:false, err: err.toString()});
  }
}

function out(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
function doGet(e) { return doPost(e); }
